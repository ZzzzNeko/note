# 函数加工

## 防抖

```js
// 防抖
function debounce(func, wait) {
  let timer = null;
  return function () {
    if (timer !== null) clearTimeout(timer);
    timer = setTimeout(() => {
      func.apply(this, arguments);
      timer = null;
    }, wait);
  };
}
```

## 节流

```js
// 节流
function throttle(func, wait) {
  let prev = 0;
  return function () {
    const now = +new Date();
    if (now - prev > wait) {
      func.apply(this, arguments);
      prev = now;
    }
  };
}
```

## 柯里化

```js
// 柯里化
function currying(fn, length) {
  const len = length || fn.length;
  return function (...args) {
    return args.length >= len
      ? fn.apply(this, args)
      : currying(fn.bind(this, ...args), len - args.length);
  };
}
```
