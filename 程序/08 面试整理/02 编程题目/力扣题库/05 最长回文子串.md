# 最长回文子串

## 题目大纲

给你一个字符串 s，找到 s 中最长的回文子串。

## 个人提交

```js
// 以前实现的
function getLongestPlalindrome(s) {
  const string = "^" + [...s].join("#") + "$";
  let longest = "";
  for (let i = 1; i < string.length - 1; i++) {
    let strts = string[i];
    let len = strts === "#" ? 0 : 1;
    let r = 1;
    while (string[i - r] === string[i + r]) {
      strts = `${string[i - r]}${strts}${string[i + r]}`;
      if (string[i - r] !== "#") len += 2;
      r++;
    }
    longest = longest.length < len ? strts.replace(/#/g, "") : longest;
  }
  return longest;
}

const str = "ccb";
console.log(getLongestPlalindrome(str));
```
