# 括号匹配

思路：栈

```js
// 给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串，判断字符串是否有效。
function isBracketValid(str) {
  const stack = [];
  const isMatch = (a, b) =>
    (a == "(" && b == ")") || (a == "{" && b == "}") || (a == "[" && b == "]");
  const getLast = arr => arr[arr.length - 1];
  for (const ele of str) {
    if (stack.length && isMatch(getLast(stack), ele)) {
      stack.pop();
    } else {
      stack.push(ele);
    }
  }
  return stack.length == 0;
}
```
