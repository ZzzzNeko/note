# 模拟 Event

```js
class MyEvent {
  constructor() {
    this.store = new Map();
  }
  on(evtName, callback) {
    if (this.store.has(evtName)) {
      const listeners = this.store.get(evtName);
      listeners.push(callback);
    } else {
      const listeners = [callback];
      this.store.set(evtName, listeners);
    }
  }
  un(evtName) {
    this.store.delete(evtName);
  }
  emit(evtName, ...args) {
    const listeners = this.store.get(evtName);
    if (!listeners) return console.warn("事件不存在");
    for (const callback of listeners) {
      callback(...args);
    }
  }
}

const myE = new MyEvent();

myE.on("show", (a, b, c) => {
  console.log(a, b, c);
});

myE.on("show", (a, b, c) => {
  console.log(a ** a, b ** b, c ** c);
});

myE.emit("show", 1, 2, 3);

myE.un("show");
myE.emit("show", 1, 2, 3);
```
