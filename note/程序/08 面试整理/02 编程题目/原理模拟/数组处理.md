## 扁平

```js
// ==================== 数组扁平 ====================
// flat-原生
function flatten(array) {
  return array.flat(Infinity);
}

// flat-递归
function flatten(array) {
  return array.reduce(
    (acc, cur) => acc.concat(Array.isArray(cur) ? flatten(cur) : cur),
    []
  );
}

// flat-迭代
function flatten(array) {
  const result = [];
  const stack = [...array];
  do {
    const head = stack.shift();
    if (Array.isArray(head)) {
      stack.unshift(...head);
    } else {
      result.push(head);
    }
  } while (stack.length);
  return result;
}
```

## 去重

```js
// unique-集合
function unique(array) {
  return [...new Set([...array])];
}

// unique-手写
function unique(array) {
  const result = [];
  let index = 0;
  do {
    if (!result.includes(array[index])) {
      result.push(array[index]);
    }
  } while (++index < array.length);
  return result;
}
```

## 遍历

```js
function ensureFn(fn) {
  if (typeof fn !== "function") {
    throw new TypeError(fn + " is not a function");
  }
}

Array.prototype.forEach = function (callback, thisArg) {
  ensureFn(callback);
  const len = this.length >>> 0;
  for (let i = 0; i < len; i++) {
    callback.call(thisArg, this[i], i, this);
  }
};

Array.prototype.filter = function (callback, thisArg) {
  ensureFn(callback);
  const len = this.length >>> 0;
  const result = [];
  for (let i = 0; i < len; i++) {
    if (callback.call(thisArg, this[i], i, this)) result.push(this[i]);
  }
  return result;
};

Array.prototype.map = function (callback, thisArg) {
  ensureFn(callback);
  const len = this.length >>> 0;
  const result = [];
  for (let i = 0; i < len; i++) {
    result.push(callback.call(thisArg, this[i], i, this));
  }
  return result;
};

Array.prototype.reduce = function (callback, initVal) {
  ensureFn(callback);
  const len = this.length >>> 0;
  for (let i = 0; i < len; i++) {
    initVal = callback(initVal, this[i], i, this);
  }
  return initVal;
};
```
