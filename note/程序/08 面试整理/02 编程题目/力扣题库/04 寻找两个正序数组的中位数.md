# 寻找两个正序数组的中位数

## 题目大纲

给定两个大小分别为 m 和 n 的正序（从小到大）数组  nums1 和  nums2。请你找出并返回这两个正序数组的 中位数 。

算法的时间复杂度应该为 O(log (m+n)) 。

var findMedianSortedArrays = function (nums1, nums2) {
  const getMid = arr => {
    if (len % 2 === 0) {
      return (arr[len / 2 - 1] + arr[len / 2]) / 2;
    } else {
      return arr[(len - 1) / 2];
    }
  };
  const len = nums1.length + nums2.length;
  const arr = [];
  let a = 0;
  let b = 0;
  if (nums1.length === 0 && nums2.length === 0) return;
  if (nums1.length === 0 && nums2.length > 0) return getMid(nums2);
  if (nums1.length > 0 && nums2.length === 0) return getMid(nums1);
  for (let i = 0; i < len; i++) {
    const x = nums1[a];
    const y = nums2[b];
    if (x === undefined) {
      arr.push(y);
      b++;
    } else if (y === undefined) {
      arr.push(x);
      a++;
    } else if (x <= y) {
      arr.push(x);
      a++;
    } else {
      arr.push(y);
      b++;
    }
  }
  return getMid(arr);
};
