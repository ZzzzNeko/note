# 斐波那契

```js
// 使用递归
function fib(n, a, b) {
  if (n <= 1) return a;
  return fib(n - 1, b, a + b);
}

// console.log(fib(7, 0, 1)); // 0 1 1 2 3 5 8

// 不使用递归
function fib(n) {
  if (n == 1) return 0;
  if (n == 2) return 1;
  let a = 0;
  let b = 1;
  let i = 3;
  let r;
  while (i <= n) {
    r = a + b;
    a = b;
    b = r;
    i++;
  }
  return r;
}
```
