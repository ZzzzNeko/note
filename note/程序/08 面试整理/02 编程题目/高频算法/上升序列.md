# 上升序列

## 给定 n ，输出所有和为 n 的连续正数序列

```js
/**
 * 给定 n ，输出所有和为 n 的连续正数序列
 * @param { number } n
 */
function getArr(n) {
  const result = [];
  const temp = [];
  const getSum = arr => arr.reduce((acc, cur) => (acc += cur), 0); // NOTE: 这里应该优化为 (head + tail) * n/2
  const array = Array((n >> 1) + 2)
    .fill(undefined)
    .map((e, i) => i + 1);
  for (const item of array) {
    if (getSum(temp) < n) {
      temp.push(item);
      while (getSum(temp) > n) {
        temp.shift();
      }
      continue;
    }
    if (getSum(temp) == n) {
      result.push([...temp]);
      temp.push(item);
      while (getSum(temp) > n) {
        temp.shift();
      }
    }
  }
  return result;
}

console.log(getArr(21));
```
